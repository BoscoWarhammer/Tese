\begin{algorithm}
    \SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
    \KwIn{body position}
    \KwOut{neighbor list}
    Divide the space in regions\;
    \ForEach{body}{
        Insert the body in the list of the region it belongs to\;
        Insert the body into adjacent lists of the region it belongs to\;
    }
